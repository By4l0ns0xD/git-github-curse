git & github curse - Sean

comands:

git init          // iniciar git
git add -A          // guardar todos los archivos de la carpeta
git commit -m "texto"          // crear commit
git status          // estado
git log         // log
git config alias.nombre-alias "comando"         // agrega un alias para comandos
git commit -am "texto"          // atajo para usar add -A y -m a la vez
git branch          // lista de branch
git branch nombre         // crear branch
git branch -d nombre         // eliminar branch
git checkout nombre-branch        // cambiar de branch
git merge nombre-branch         // funcionar branch
git branch --merged         // saber que branch estan funcionados
o       // para que deje escribir(--INSERT--) commit presionar " o "
ESC         // para permitir salir de funciones (commit)
:x          // guarda y sale de commit funciones rapidas

git branch --help         // ayuda sobre branch
git log --help         // ayuda sobre log
